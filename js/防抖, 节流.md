#### 防抖

防抖可以概括为触发高频事件后 n 秒内函数只会执行一次，如果 n 秒内高频事件再次被触发，则重新计算时间

```javascript
// 延迟执行
function debounce(fn, time) {
  let timer = null;

  return () => {
    // 每次都会先清除旧的计时器, 重新创建新的计时器
    clearTimeout(timer);
    timer = setTimeout(() => {
      fn();
    }, time);
  };
}

// 立即执行
function debounce(fn, time) {
  let timer = null;

  return () => {
    // 当计时器存在时, 清空当前计时器
    if (timer) {
      clearTimeout(timer);
    } else {
      // 如果计时器不存在, 才执行函数
      fn();
    }
    // 不管计时器是否存在, 都重新创建新的计时器覆盖旧的, 以此达到刷新计时的目的
    timer = setTimeout(() => {
      clearTimeout(timer);
      timer = null;
    }, time);
  };
}
```

#### 节流

节流就是减少流量，将频繁触发的事件减少，并每隔一段时间执行。即，控制事件触发的频率

```javascript
// 延迟执行
function throttle(fn, time) {
  let timer = null;
  return () => {
    // 仅当计时器完成时执行函数, 并创建新的计时器
    if (!timer) {
      timer = setTimeout(() => {
        fn();
        clearTimeout(timer);
        timer = null;
      }, time);
    }
  };
}
// 立即执行
function throttle(fn, time) {
  let timer = null;
  return () => {
    // 仅当计时器完成时才执行函数, 并创建新的计时器
    if (!timer) {
      // 先执行函数
      fn();
      // 再创建计时器
      timer = setTimeout(() => {
        clearTimeout(timer);
        timer = null;
      }, time);
    }
  };
}
```
